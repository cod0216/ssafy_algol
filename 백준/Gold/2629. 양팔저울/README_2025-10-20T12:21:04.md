import java.util.*;
import java.io.*;

public class Main {
    static int N;
    static int[] arr, A;
    static boolean[][] dp;
    public static void main(String[] args) throws IOException {  
        BufferedReader buffer = new BufferedReader(new InputStreamReader(System.in));
        StringBuilder sb = new StringBuilder();
        N = Integer.parseInt(buffer.readLine());
        arr = new int[N+1];
        dp = new boolean[31][15001];
        String[] input = buffer.readLine().split(" ");
        for(int i = 1; i <= N; i++){
           arr[i] = Integer.parseInt(input[i-1]);
        }
        
        int M = Integer.parseInt(buffer.readLine());
        A = new int[M];
        
        dp(0, 0);
        
        input = buffer.readLine().split(" ");
        for(int i = 0; i < M; i++){
           int v = Integer.parseInt(input[i]);
            if(v > 15000) sb.append("N ");
            else{
                sb.append(dp[N][v] ? "Y " : "N ");
            }
        } 
        
        System.out.println(sb.toString());
    }
    
    public static void dp(int idx, int w){
        if(dp[idx][w]) return;
        
        dp[idx][w] = true;
        if(idx == N) return;
        
        dp(idx+1, w);
        dp(idx+1, w+arr[idx+1]);
        dp(idx+1, Math.abs(w-arr[idx+1]));
    }
}
